            return;

                date,
                time,
                duration: Number(duration),
                description: description.trim() || undefined,

                setCreatedMeeting(null);
                resetForm();

                setMeetingId(response.meeting.id);

                resetForm({ preserveMeetingId: true });

            await loadMeetings();

            setIsCreating(false);
        }
